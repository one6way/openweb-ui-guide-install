# По одной цели на строку. Комментарии начинаются с '#'. Пустые строки игнорируются.
# Формат: host[:port][,sni=example.com][,days=NN]

# Примеры:
# localhost со стандартным портом 443
localhost

# Явный порт
# my.internal.service:8443

# IP с переопределением SNI (SNI=публичное имя из сертификата)
# 10.0.12.34:443,sni=api.example.com

# Более жёсткий порог для конкретного хоста
# staging.example.com:443,days=10
